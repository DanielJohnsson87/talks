```javascript
import { configureStore } from "@reduxjs/toolkit";

function vehiclesReducer(state = [], action) {
	switch(action.type) {
		case "VEHICLE_ADDED":
			return [...state, action.payload];
	    default:
			return state
	}
}

const store = configureStore({
	reducer: vehiclesReducer
});

store.dispatch({ 
	type: "VEHICLE_ADDED",
	payload: { id: 1, name: "Vehicle 1" }
})
```
